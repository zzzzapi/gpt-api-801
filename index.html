# Python调用GPT API示例
import requests

API_KEY = "your_4z_api_key"
API_URL = "https://api.zzzzapi.com/v1/chat/completions"

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "model": "gpt-4o",
    "messages": [
        {"role": "system", "content": "你是一位AI助手"},
        {"role": "user", "content": "如何使用GPT API开发一个聊天机器人?"}
    ],
    "temperature": 0.7
}

response = requests.post(API_URL, headers=headers, json=data)
print(response.json())
        

// JavaScript/Node.js调用GPT API示例
const axios = require('axios');

const API_KEY = 'your_4z_api_key';
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

async function callGptApi() {
  try {
    const response = await axios.post(API_URL, {
      model: 'gpt-4o',
      messages: [
        {role: 'system', content: '你是一位AI助手'},
        {role: 'user', content: '如何使用GPT API开发一个聊天机器人?'}
      ],
      temperature: 0.7
    }, {
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json'
      }
    });
    
    console.log(response.data);
  } catch (error) {
    console.error('API调用出错:', error.response?.data || error.message);
  }
}

callGptApi();
